#!/bin/bash

# 指定音频文件路径
AUDIO_FILE="~/Music/your-audio-file.mp3"

# 检查音频文件是否存在
if [ -f "$AUDIO_FILE" ]; then
    echo "正在播放：$AUDIO_FILE"
    mpv "$AUDIO_FILE"  # 或者使用 vlc "$AUDIO_FILE"
else
    echo "音频文件未找到：$AUDIO_FILE"
fi
